program factorial;


int function factorial(n: int)
{
    var int fact;
    fact = 0;

    if (n < 2)
    {
        return 1;
    }
    else
    {
        return n * factorial(n - 1);
    }
}

function main()
{
    var int x, y;
    output("Factorial of what number?");
    input(x);
    y = factorial(x);
    output(y);
}
